\node (TimeHierarchie) [satz,text width=\satzW, minimum width=\satzWmin, anchor=south east] 
at ([xshift=-1.5cm,yshift=5.5cm] current page.south east) 
{
	\textbf{Time hierarchy theorem} \hfill Hartmanis, Stearns  \nodepart{second} 
	For  time constructible $f,g$ with 
	$f(n)\in\Omega(n)$ and $g(n)\in\omega(f(n)\cdot\log (f(n)))$ holds: \[\mathbf{TIME(f)\subsetneq TIME(g)}\]
};

\node (SpaceHierarchie) [satz2,text width=\satzW, minimum width=\satzWmin, anchor=south east] 
at (TimeHierarchie.north east) 
{
	\textbf{Space hierarchy theorem} \hfill Stearns, Szepietowski  \nodepart{second} 
	For  time constructible $f,g$ with 
	$f(n)\in\Omega(\log n)$ and $g(n)\in\omega(f(n))$ holds:
	\[\mathbf{SPACE(f)\subsetneq SPACE(g)}\]
	\[\mathbf{NSPACE(f)\subsetneq NSPACE(g)}\]
};

\node (SatzSavitch) [satz,text width=\satzW, minimum width=\satzWmin, anchor=south east] 
at (SpaceHierarchie.north east) 
{
	\textbf{Savitch's theorem} \hfill Savitch  \nodepart{second} 
	For space constructible $S$  with $S(n)\geq \log n$ holds:
	$$ \mathbf{NSPACE(S)\subseteq SPACE(S^2)}$$
};

\node (SatzImmerman) [satz2,text width=\satzW, minimum width=\satzWmin, anchor=south east] 
at (SatzSavitch.north east) 
{
	\textbf{Immerman–Szelepcsényi theorem} 
%	\hfill Immerman, Szelepcsényi  
	\nodepart{second}
	For space constructible $S$ with $S(n)\geq\log n$ holds:
	$$ \mathbf{NSPACE(S)}=\mathbf{coNSPACE(S)} $$
};

\node (SatzAlternation) [satz,text width=\satzW, minimum width=\satzWmin, anchor=south east] 
at (SatzImmerman.north east) 
{
	\textbf{Theorem} \hfill Chandra, Kozen, Stockmeyer   \nodepart{second} 
	\begin{enumerate}[(a)]
	\item For $S(n)=\Omega(\log n)$ holds: $\mathbf{ASPACE(S)\subseteq TIME(2^{\mathcal{O}(S)})}$
	\item For $T(n)=\Omega(n)$ holds: $\mathbf{TIME(T)\subseteq ASPACE(\log T)}$

	\textbf{PCP} \textit{(probabilistically checkable proof)} \nodepart{second} 
$PCP(\mathcal{F},\mathcal{G})$ is the class of languages $L$ such there is a verifier $V$ that
\begin{itemize}
	\item uses $\leq r(n)$ random bits, \hfill($r\in\mathcal{F}$)
	\item reads $\leq q(n)$ bits of the proof, \hfill($q\in\mathcal{G}$)
	\item accepts for a proof $\pi$ everything from~$L$
